export function reduceOperator(reducer, seed, forced) {
  if (isUndefined(reducer)) {
    reducer = identity;
    seed = constant();
  }
  else if (isFunction(reducer)) seed = toFunction(seed);
  else return tie(valueAdapter, reducer);
