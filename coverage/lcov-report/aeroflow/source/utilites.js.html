export const toDelay = (value, def) => {
  switch (classOf(value)) {
    case DATE:
      value = value - dateNow();
      break;
    case NUMBER:
      break;
    default:
      value = +value;
      break;
  }
  return isNaN(value) ? def : value < 0 ? 0 : value;
}
